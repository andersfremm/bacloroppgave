group -n mac -t bank -i {TxFifo RxFifo TxCnt RxCnt TxFifoFull TxFifoEmpty RxFifoFull RxFifoEmpty LoopEn TxStart FifoRst}

group -n mac -gw {12 8}

reg -n TxFifo       -gtype entry

reg -n TxFifoFull   -gloc {0  4}

reg -n TxFifoEmpty  -gloc {0  5}

reg -n LoopEn       -gloc {0  8}
